```typescript
import { useState, useEffect, useCallback } from 'react';

const API_BASE = '/api/alerts';

export interface Alert {
  id: number;
  projectId: string;
  projectName: string;
  type: 'overdue' | 'budget_limit';
  message: string;
  severity: 'warning' | 'critical';
  details: Record<string, unknown>;
}

export interface AlertsResponse {
  alerts: Alert[];
  count: number;
  timestamp: string;
}

// Fetch alerts
export async function fetchAlerts(): Promise<AlertsResponse> {
  const response = await fetch(API_BASE);
  if (!response.ok) {
    throw new Error('Failed to fetch alerts');
  }
  return response.json();
}

// Custom hook for polling alerts
export function useAlerts(pollingInterval = 60000) {
  const [data, setData] = useState<AlertsResponse | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const refresh = useCallback(async () => {
    try {
      setError(null);
      const result = await fetchAlerts();
      setData(result);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Failed to load alerts');
    } finally {
      setLoading(false);
    }
  }, []);

  // Initial load
  useEffect(() => {
    refresh();
  }, [refresh]);

  // Polling
  useEffect(() => {
    if (pollingInterval <= 0) return;

    const interval = setInterval(refresh, pollingInterval);
    return () => clearInterval(interval);
  }, [refresh, pollingInterval]);

  return {
    alerts: data?.alerts || [],
    count: data?.count || 0,
    loading,
    error,
    refresh,
  };
}

// Alert type labels
export const ALERT_TYPE_LABELS: Record<string, string> = {
  overdue: 'Overdue',
  budget_limit: 'Budget Alert',
};

// Alert type icons (for use with lucide-react)
export const ALERT_TYPE_ICONS: Record<string, string> = {
  overdue: 'Calendar',
  budget_limit: 'DollarSign',
};

// Severity colors
export const SEVERITY_COLORS: Record<string, string> = {
  warning: 'text-yellow-600 bg-yellow-50',
  critical: 'text-red-600 bg-red-50',
};
```

```tsx
import { useState } from 'react';
import { Button } from '@/components/ui/button';
import { Badge } from '@/components/ui/badge';
import {
  Popover,
  PopoverContent,
  PopoverTrigger,
} from '@/components/ui/popover';
import {
  useAlerts,
  ALERT_TYPE_LABELS,
  SEVERITY_COLORS,
  type Alert,
} from '@/lib/alerts-api';
import { Bell, Calendar, DollarSign, RefreshCw, AlertTriangle } from 'lucide-react';

interface AlertsDropdownProps {
  onAlertClick?: (projectId: number) => void;
}

export function AlertsDropdown({ onAlertClick }: AlertsDropdownProps) {
  const [open, setOpen] = useState(false);
  const { alerts, count, loading, error, refresh } = useAlerts(60000);

  const handleAlertClick = (alert: Alert) => {
    setOpen(false);
    onAlertClick?.(alert.id);
  };

  return (
    <Popover open={open} onOpenChange={setOpen}>
      <PopoverTrigger asChild>
        <Button variant="ghost" size="sm" className="relative">
          <Bell className="h-5 w-5" />
          {count > 0 && (
            <Badge
              variant="destructive"
              className="absolute -top-1 -right-1 h-5 min-w-5 flex items-center justify-center p-0 text-xs"
            >
              {count > 99 ? '99+' : count}
            </Badge>
          )}
        </Button>
      </PopoverTrigger>

      <PopoverContent className="w-80 p-0" align="end">
        {/* Header */}
        <div className="flex items-center justify-between p-3 border-b">
          <h4 className="font-semibold text-sm">Alerts</h4>
          <Button
            size="sm"
            variant="ghost"
            onClick={refresh}
            disabled={loading}
          >
            <RefreshCw className={`h-4 w-4 ${loading ? 'animate-spin' : ''}`} />
          </Button>
        </div>

        {/* Content */}
        <div className="max-h-80 overflow-auto">
          {error ? (
            <div className="p-4 text-center text-sm text-destructive">
              {error}
            </div>
          ) : alerts.length === 0 ? (
            <div className="p-6 text-center">
              <Bell className="h-8 w-8 mx-auto text-muted-foreground/50 mb-2" />
              <p className="text-sm text-muted-foreground">No active alerts</p>
            </div>
          ) : (
            <div className="divide-y">
              {alerts.map((alert) => (
                <AlertItem
                  key={`${alert.type}-${alert.id}`}
                  alert={alert}
                  onClick={() => handleAlertClick(alert)}
                />
              ))}
            </div>
          )}
        </div>
      </PopoverContent>
    </Popover>
  );
}

// Individual alert item
function AlertItem({
  alert,
  onClick,
}: {
  alert: Alert;
  onClick: () => void;
}) {
  const Icon = alert.type === 'overdue' ? Calendar : DollarSign;
  const severityClass = SEVERITY_COLORS[alert.severity] || '';

  return (
    <button
      className="w-full p-3 text-left hover:bg-muted/50 transition-colors"
      onClick={onClick}
    >
      <div className="flex gap-3">
        {/* Icon */}
        <div className={`shrink-0 p-2 rounded-full ${severityClass}`}>
          {alert.severity === 'critical' ? (
            <AlertTriangle className="h-4 w-4" />
          ) : (
            <Icon className="h-4 w-4" />
          )}
        </div>

        {/* Content */}
        <div className="flex-1 min-w-0">
          <div className="flex items-center gap-2">
            <span className="font-medium text-sm truncate">
              {alert.projectId}
            </span>
            <Badge
              variant="outline"
              className={`text-xs shrink-0 ${
                alert.severity === 'critical'
                  ? 'border-red-300 text-red-700'
                  : 'border-yellow-300 text-yellow-700'
              }`}
            >
              {ALERT_TYPE_LABELS[alert.type] || alert.type}
            </Badge>
          </div>
          <p className="text-sm text-muted-foreground truncate">
            {alert.projectName}
          </p>
          <p className="text-xs text-muted-foreground mt-1">
            {alert.message}
          </p>
        </div>
      </div>
    </button>
  );
}
```
