- Returns 400 for invalid payload (missing name)
- Returns 409 for duplicate name

Test cases for DELETE /api/admin/departments/:id:
- Returns 204 for successful deletion
- Returns 404 for non-existent id
- Returns 409 when department has teams (usage > 0)

Example test structure:
```typescript
describe('Departments API', () => {
  let app: FastifyInstance

  beforeAll(async () => {
    app = await build({ logger: false })
  })

  beforeEach(async () => {
    await seedTestData(app.db)
  })

  afterEach(async () => {
    await clearTestData(app.db)
  })

  afterAll(async () => {
    await app.close()
  })

  test('GET /api/admin/departments returns list', async () => {
    const response = await app.inject({
      method: 'GET',
      url: '/api/admin/departments'
    })
    expect(response.statusCode).toBe(200)
    expect(response.json()).toBeInstanceOf(Array)
  })
})
```

Create backend/src/lib/cost-tshirt.test.ts:
- Test `deriveCostTshirt(totalBudget, thresholds)` function
- Mock thresholds array as input (no database needed)

Test cases:
- Returns 'XS' for budget below smallest threshold
- Returns 'S' for budget in S range
- Returns 'M' for budget in M range
- Returns 'L' for budget in L range
- Returns 'XL' for budget in XL range
- Returns 'XXL' for budget above largest threshold
- Returns null for null/undefined budget
- Handles edge cases at exact threshold boundaries

```typescript
describe('deriveCostTshirt', () => {
  const mockThresholds = [
    { size: 'XS', maxAmount: '10000' },
    { size: 'S', maxAmount: '50000' },
    { size: 'M', maxAmount: '100000' },
    { size: 'L', maxAmount: '250000' },
    { size: 'XL', maxAmount: '500000' },
    { size: 'XXL', maxAmount: null }, // Above XL
  ]

  test('returns XS for small budgets', () => {
    expect(deriveCostTshirt(5000, mockThresholds)).toBe('XS')
  })

  test('returns null for null budget', () => {
    expect(deriveCostTshirt(null, mockThresholds)).toBeNull()
  })
})
```
